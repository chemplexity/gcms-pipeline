create table peaks
(
    peak_id      integer not null
        constraint peak_id
            primary key,
    peak_time    TEXT    not null,
    peak_area    REAL    not null,
    peak_width   REAL    not null,
    peak_height  REAL    not null,
    area_Of      REAL,
    error        REAL,
    model        TEXT,
    x_min        REAL, 
    x_max        REAL, 
    y_min        REAL, 
    y_max        REAL,
    input_x      TEXT,
    input_y      TEXT,
    date_created TEXT,
    peak_mz      TEXT,
    peak_intens  TEXT,
    sample_id    integer
        constraint sample_id
            references samples
);
